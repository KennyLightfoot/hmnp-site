# Notification System Cron Jobs
# Copy these entries into your server's crontab using: crontab -e

# Environment variables
APP_URL=https://your-production-domain.com
SCHEDULER_SECRET=your-secret-key-here
NODE_PATH=/path/to/your/node/executable

# Initialize scheduler every 6 hours (to handle server restarts)
0 */6 * * * cd /path/to/your/app && $NODE_PATH scripts/trigger-notifications.js --mode=init >> /var/log/hmnp-scheduler-init.log 2>&1

# Run short-term reminders check every 5 minutes
*/5 * * * * cd /path/to/your/app && $NODE_PATH scripts/trigger-notifications.js --mode=check >> /var/log/hmnp-scheduler-check.log 2>&1

# Check scheduler status once a day (for monitoring)
0 9 * * * cd /path/to/your/app && $NODE_PATH scripts/trigger-notifications.js --mode=status >> /var/log/hmnp-scheduler-status.log 2>&1
